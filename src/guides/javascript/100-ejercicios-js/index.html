Copiar al portapapeles

const copy_clipboard = (text) => {
  if (typeof text != "string") {
    throw TypeError("El argumento debe ser un string");
  }
  const areaText = document.createElement("textarea");
  areaText.value = text;
  areaText.setAttribute("readonly", "");
  areaText.style.position = "absolute";
  areaText.style.left = "-9999px";
  document.body.appendChild(areaText);
  const selection =
    document.getSelection().rangeCount > 0
      ? document.getSelection().getRangeAt(0)
      : false;

  areaText.select();

  document.execCommand("copy");
  document.body.removeChild(areaText);

  if (selection) {
    document.getSelection().removeAllRanges();
    document.getSelection().addRange(selection);
  }
};

Saber si una propieda existe en un objeto

Para saber si un objeto tiene una propiedad utilizamos el método .hasOwnProperty, el cual recibe como parámetro el nombre de la propiedad sobre la cuál queremos conocer su existencia.

class Rectangulo {
  constructor(alto, ancho) {
    this.alto = alto;
    this.ancho = ancho;
  }
  // Getter
  get area() {
    return this.calcularArea();
  }
  // Método
  calcularArea() {
    return this.alto * this.ancho;
  }
}

const cuadrado = new Rectangulo(10, 10);

if(cuadrado.hasOwnProperty("ancho")) {
  console.log("Si existe")
}
else {
  console.log("No existe la propiedad");
}
